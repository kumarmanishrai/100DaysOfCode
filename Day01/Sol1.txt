var diagonalSum = function(mat) {

    let sum = 0;
    let n = mat[0].length
    for(var i=0; i< n; i++){
        sum = sum + mat[i][i]
    }
    for(var i = 0; i<n; i++){
        if(i == n-1-i){
            continue 
        }
        sum = sum + mat[i][n-1-i]
    } 
    return sum
};
